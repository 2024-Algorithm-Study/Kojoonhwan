from collections import deque

def move(board, r, c, direction):
    # 0: up
    # 1: right
    # 2: down
    # 3: left
    if direction == 0:
        while r >= 0 and board[r][c] != 'D':
            r -= 1
        r += 1
    elif direction == 1:
        while c < len(board[0]) and board[r][c] != 'D':
            c += 1
        c -= 1
    elif direction == 2:
        while r < len(board) and board[r][c] != 'D':
            r += 1
        r -= 1
    else:
        while c >= 0 and board[r][c] != 'D':
            c -= 1
        c += 1

    return r, c
